Equalize the Array

// Complete the equalizeArray function below.
int equalizeArray(vector<int> arr) {

    sort(arr.begin(), arr.end());
    unordered_map<int,int> repeats;

    for (int i=0;i<arr.size()-1;i++) {
        if (arr[i] == arr[i+1]) {
            if (repeats.count(arr[i]) == 0) {
                repeats[arr[i]] = 2;
            } else {
                repeats[arr[i]]++;
            }
        }
    }

    int max = 1;

    for (const auto& val : repeats) {
        max = (val.second > max) ? val.second : max;
    }

    return arr.size() - max;
}

Solution in Go:

// Complete the equalizeArray function below.
func equalizeArray(arr []int32) int32 {

    repeats := make(map[int32]int)
    max := 1

    sort.Slice(arr, func(i, j int) bool { return arr[i] < arr[j] })

    for i := 0; i < len(arr)-1; i++ {
        if arr[i] == arr[i+1] {

            if _, exist := repeats[arr[i]]; exist {

                repeats[arr[i]]++
            } else {
                repeats[arr[i]] = 2
            }
        }
    }

    //find max value in map
    for _, v := range repeats {
        if v > max {
            max = v
        }
    }

    return int32(len(arr) - (max))
}
